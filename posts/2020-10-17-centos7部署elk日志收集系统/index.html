Centos7部署ELK日志收集系统

Posted by on Sat, Oct 17, 2020

 

一、安装Elasticsearch

1、下载安装包并解压

[root@ELK ~]# wget https://artifacts.elastic.co/downloads/elasticsearch/elasticsearch-7.9.2-linux-x86_64.tar.gz [root@ELK ~]# tar xzvf elasticsearch-7.9.2-linux-x86_64.tar.gz -C /usr/local/

2、创建用户和用户组

elasticsearch默认不允许使用root用户运行，如果要强行使用root运行，要加上该参数 ./bin/elasticsearch -Des.insecure.allow.root=true

[root@ELK ~]# useradd es [root@ELK ~]# passwd es [root@ELK ~]# chown -R es:es /usr/local/elasticsearch-7.9.2/

3、启动elasticsearch

[root@ELK ~]# su - es [es@ELK ~]# cd /usr/local/elasticsearch-7.9.2/ [es@ELK elasticsearch-7.9.2]$ ./bin/elasticsearch [root@ELK ~]# curl http://127.0.0.1:9200 #测试启动是否成功

启动出错

[root@ELK ~]# tail -f /var/log/messages #需要设置es用户的最大打开文件数和最大进程数 [1]: max file descriptors [4096] for elasticsearch process is too low, increase to at least [65535]

#需要修改单个进程拥有虚拟内存区域的大小 [2]: max virtual memory areas vm.max_map_count [65530] is too low, increase to at least [262144]

#需要配置elasticsearch.yml文件，指定discovery主机 [3]: the default discovery settings are unsuitable for production use; at least one of [discovery.seed_hosts, discovery.seed_providers, cluster.initial_master_nodes] must be configured

4、配置系统内核参数

修改es用户的最大打开文件数和最大进程数

[root@ELK ~]# vim /etc/security/limits.conf //在文件最后加入 es soft nofile 65536 es hard nofile 65536 es soft nproc 4096 es hard nproc 4096

修改单个进程拥有虚拟内存区域的大小

[root@ELK ~]# vim /etc/sysctl.conf vm.max_map_count=655360

重启系统生效

[root@ELK ~]# reboot

5、配置elasticsearch

[root@ELK ~]# vim /usr/local/elasticsearch-7.9.2/config/elasticsearch.yml network.host: 192.168.1.225 #修改为本机IP，允许其它主机远程访问 discovery.seed_hosts: [“192.168.1.225”] #集群主机节点，单节点则只填本机 #path.data: /path/to/data #数据存放目录，默认在./elasticsearch/data下 #path.logs: /path/to/logs #日志存放目录，默认在./elasticsearch/logs下

6、配置防火墙

[root@ELK ~]# firewall-cmd –permanent –add-port=9200/tcp [root@ELK ~]# firewall-cmd –reload

7、配置为服务并开机启动

[root@ELK ~]# vim /etc/systemd/system/elasticsearch.service [Unit] Description=elasticsearch.service After=network.target

[Service] Type=forking User=es #使用哪个用户启动

LimitNOFILE=100000 LimitNPROC=100000 ExecStart=/usr/local/elasticsearch-7.9.2/bin/elasticsearch -d #启动的程序路径

[Install] WantedBy=multi-user.target

启动

[root@ELK ~]# systemctl enable elasticsearch [root@ELK ~]# systemctl start elasticsearch

二、安装插件elasticsearch-head

1、安装git和nodejs（nodejs需要9.0或以上版本）

[root@ELK ~]# yum -y install git [root@ELK ~]# wget https://npm.taobao.org/mirrors/node/v14.13.1/node-v14.13.1-linux-x64.tar.xz [root@ELK ~]# tar xf node-v14.13.1-linux-x64.tar.xz -C /usr/local/ [root@ELK ~]# mv /usr/local/node-v14.13.1-linux-x64/ /usr/local/nodejs [root@ELK ~]# ln -sv /usr/local/nodejs/bin/npm /usr/local/bin/ [root@ELK ~]# ln -sv /usr/local/nodejs/bin/node /usr/local/bin/

2、安装配置elasticsearch-head

[root@ELK ~]# cd /usr/local/ [root@ELK local]# git clone git://github.com/mobz/elasticsearch-head.git [root@ELK local]# cd elasticsearch-head/ [root@ELK elasticsearch-head]# npm install

#npm默认源安装较慢，更换淘宝源加快安装速度 [root@ELK ~]# npm config set registry https://registry.npmjs.org/ #配置官方的源 [root@ELK ~]# npm config set registry https://registry.npm.taobao.org #配置淘宝的源 [root@ELK ~]# npm config get registry #查看当前注册的源

npm install出错，查看npm日志

[root@ELK elasticsearch-head]# less /root/.npm/_logs/2020-10-03T17_03_29_020Z-debug.log …….. Failed at the phantomjs-prebuilt@2.1.16 install script.

解决方法，单独安装phantomjs：

[root@ELK elasticsearch-head]# npm install phantomjs-prebuilt@2.1.14 –ignore-scripts

3、配置elasticsearch允许跨域访问

[root@ELK ~]# vim /usr/local/elasticsearch-7.9.2/config/elasticsearch.yml

//在文件末尾追加下面两行代码 http.cors.enabled: true http.cors.allow-origin: “*”

4、修改elasticsearch-head默认连接地址

[root@ELK ~]# vim /usr/local/elasticsearch-head/_site/app.js this.base_uri = this.config.base_uri || this.prefs.get(“app-base_uri”) || “http://192.168.1.225:9200”; #将localhost修改为es服务器地址

5、配置防火墙

[root@ELK ~]# firewall-cmd –permanent –add-port=9100/tcp [root@ELK ~]# firewall-cmd –reload

6、设置开机启动

方法一：配置为服务

[root@ELK ~]# vim /etc/init.d/elasticsearch-head #!/bin/sh #chkconfig: 2345 80 05 #description: elasticsearch-head

elasticsearch-head 的路径

cd /usr/local/elasticsearch-head nohup npm run start >/usr/local/elasticsearch-head/nohup.out 2>&1 &

[root@ELK ~]# chmod +x /etc/init.d/elasticsearch-head [root@ELK ~]# chkconfig –add elasticsearch-head

方法二：配置开机启动脚本

[root@ELK ~]# vim /etc/rc.local [root@ELK ~]# chmod +x /etc/rc.d/rc.local cd /usr/local/elasticsearch-head/node_modules/grunt/bin/ ./grunt server&

7、解决单节点 Elasticsearch 健康状态为 yellow 问题

原因：单点部署的 Elasticsearch，默认分片的副本数为 1，而相同的分片不能在同一个节点上，所以就出现上面 unsigned shards 的问题

[root@ELK ~]# curl -X PUT “192.168.1.225:9200/_settings” -H ‘Content-Type: application/json’ -d ‘{“number_of_replicas”:0}’ 返回 {“acknowledged”:true}

三、安装Logstash

1、安装JAVA

官网下载Java：https://www.oracle.com/java/technologies/javase/javase-jdk8-downloads.html

[root@ELK ~]# tar xvf jdk-8u261-linux-x64.tar.gz -C /usr/local/ [root@ELK ~]# mv /usr/local/jdk1.8.0_261/ /usr/local/java

[root@ELK ~]# vim /etc/profile export JAVA_HOME=/usr/local/java export JRE_HOME=${JAVA_HOME}/jre export PATH=${JAVA_HOME}/bin:$PATH

[root@ELK ~]# source /etc/profile #立即加载profile文件中的环境变量 [root@ELK ~]# java -version #测试

2、安装logstash

官网下载：https://www.elastic.co/cn/downloads/logstash

[root@ELK ~]# wget https://artifacts.elastic.co/downloads/logstash/logstash-7.9.2.tar.gz [root@ELK ~]# tar xvf logstash-7.9.2.tar.gz -C /usr/local/ [root@ELK ~]# chown -R root:root /usr/local/logstash-7.9.2/ [root@ELK ~]# cd /usr/local/logstash-7.9.2/ [root@ELK ~]# bin/logstash -e ‘input { stdin { } } output { stdout {} }’ #运行测试

#测试，将logstash收集的内容写入到elasticsearch [root@ELK ~]# /usr/local/logstash-7.9.2/bin/logstash -e ‘input { stdin{} } output { elasticsearch { hosts => “192.168.1.225:9200” } stdout{ codec => rubydebug } }’

3、logstash指定配置文件运行

新建配置文件，用于收集syslog日志，监听514端口

[root@ELK ~]# vim /root/mylogstash/logstash.conf input{ syslog { type => “system-syslog” host => “192.168.1.225” #监听本地哪个地址，默认0.0.0.0 port => “514” } } output { elasticsearch { hosts => [“192.168.1.225:9200”] } }

加载配置文件并运行logstash

[root@ELK ~]# /usr/local/logstash-7.9.2/bin/logstash -f /root/mylogstash/logstash.conf

配置防火墙

[root@ELK ~]# firewall-cmd –permanent –add-port=514/tcp [root@ELK ~]# firewall-cmd –permanent –add-port=514/udp [root@ELK ~]# firewall-cmd –reload

4、配置为服务并开机启动

[root@ELK ~]# vim /usr/local/logstash-7.9.2/config/startup.options JAVACMD=/usr/local/java/bin/java #指定java二进制文件路径 LS_HOME=/usr/local/logstash-7.9.2/ #指定logstash目录 #LS_SETTINGS_DIR=/usr/local/logstash-7.9.2/config/ #注释该行 LS_OPTS="-f /root/mylogstash/" LS_USER=root LS_GROUP=root

[root@ELK ~]# /usr/local/logstash-7.9.2/bin/system-install #生成logstash.service文件 [root@ELK ~]# cat /etc/systemd/system/logstash.service [root@ELK ~]# systemctl enable logstash [root@ELK ~]# systemctl start logstash

注意：使用logstash收集syslog日志并监听514端口时，必须使用root用户启动Logstash，因为Linux系统中对1024以下端口需要root权限

四、安装Kibana

1、安装kibana

官网：https://www.elastic.co/cn/downloads/kibana

[root@ELK ~]# wget https://artifacts.elastic.co/downloads/kibana/kibana-7.9.2-linux-x86_64.tar.gz [root@ELK ~]# tar -xvf kibana-7.9.2-linux-x86_64.tar.gz -C /usr/local/ [root@ELK ~]# mv /usr/local/kibana-7.9.2-linux-x86_64/ /usr/local/kibana-7.9.2

2、配置Kibana

[root@ELK ~]# vim /usr/local/kibana-7.9.2/config/kibana.yml server.host: “192.168.1.225” #修改为本机IP地址，允许远程用户连接 elasticsearch.hosts: [“http://192.168.1.225:9200”] #修改为elasticsearch所在服务器地址

3、前台运行Kibana

[root@ELK ~]# cd /usr/local/kibana-7.9.2/ [root@ELK kibana-7.9.2]# bin/kibana

4、配置防火墙

[root@ELK ~]# firewall-cmd –permanent –add-port=5601/tcp [root@ELK ~]# firewall-cmd –reload

5、浏览器访问：http://192.168.1.225:5601/

6、将Kibana配置成服务开机启动

[root@ELK ~]# useradd kibana -s /sbin/nologin [root@ELK ~]# chown -R kibana:kibana /usr/local/kibana-7.9.2/

[root@ELK ~]# vim /lib/systemd/system/kibana.service [Unit] Description=Kibana After=network.target

[Service] Type=simple User=kibana Group=kibana ExecStart=/usr/local/kibana-7.9.2/bin/kibana PIDFile=/var/run/kibana.pid Restart=always

[Install] WantedBy=default.target

启动

[root@ELK ~]# systemctl enable kibana.service [root@ELK ~]# systemctl start kibana.service
