import base64

value=“qeBQAAAABJ……RU5ErkJggg==” #填写base64编码 img = base64.b64decode(value) fh = open(“pic.jpg”,“wb”) fh.write(img)

将图片转换成data:image/base64

import base64

with open(“pic.jpg”,“rb”) as f: base64_data = base64.b64encode(f.read()) print(base64_data)
