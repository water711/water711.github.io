JAVA使用opencv通过RTSP流打开摄像头

Oct 29, 2019
2 min read

 

一、安装OpenCV

打开OpenCV官网，下载对应系统的opencv，并解压到任意位置

解压./opencv/build/java目录下有以下文件

 

二、Eclipse加载OpenCV库

官方配置文档：https://docs.opencv.org/2.4/doc/tutorials/introduction/java_eclipse/java_eclipse.html

1、启动eclipse并从菜单中选择 Window –> Preferences

在左边导航窗口选择  Java –> Build Path –> User Libraries，点击New，并为你的library输入一个名称，例如：OpenCV-4.1.1

2、选中新建的user libraries，点击 Add External JARs…

选择opencv的jar包

3、选中 Native library location，点击Edit

4、选择对应opencv库文件所在文件夹，32位系统选择X86，64位系统选择X64

现在已经配置好opencv了，下面新建一个java项目，测试一下opencv是否能正常使用

 

 

三、新建项目测试OpenCV

Eclipse新建项目后，右键项目，点击 Build Path–>Add Libraries…

选择User Library

勾选opencv

创建一个新的Java文件，使用以下代码运行测试：

import org.opencv.core.Core; import org.opencv.core.CvType; import org.opencv.core.Mat;

public class opencvTest { public static void main( String[] args ) { System.loadLibrary( Core.NATIVE_LIBRARY_NAME ); Mat mat = Mat.eye( 3, 3, CvType.CV_8UC1 ); System.out.println( “mat = " + mat.dump() ); } }

运行结果

mat = [ 1, 0, 0; 0, 1, 0; 0, 0, 1]

 

四、打开摄像头，按原分辨率显示

使用RTSP流媒体打开摄像头

import org.opencv.core.Mat; import org.opencv.core.Core; import org.opencv.highgui.HighGui; import org.opencv.videoio.VideoCapture;

public class opencvTest {

static {
    System.loadLibrary(Core.NATIVE\_LIBRARY\_NAME);
}

public static void main(String\[\] args) {
System.load("D:/Program Files/opencv/build/bin/opencv\_videoio\_ffmpeg411\_64.dll");   //加载opencv\_ffmpeg400\_64.dll，要不然是无法解码
	VideoCapture cap = new VideoCapture("rtsp://192.168.104.130/user=admin&password=admin&channel=1");
    while(cap.isOpened()){
        boolean flag;
        Mat image = new Mat();
        flag = cap.read(image);
        if(!flag){
            System.out.println("--- not able to get image ---");
            break;
        }
        HighGui.imshow("video", image);
        HighGui.waitKey(10);
    }
    System.out.println("--- video capture closed ---");

} }

 

五、打开摄像头，显示在固定大小的窗口

 

import org.opencv.core.Mat; import org.opencv.core.Core; import org.opencv.videoio.VideoCapture; import org.opencv.core.Size; import org.opencv.highgui.HighGui; import org.opencv.imgproc.Imgproc;

public class opencvTest {

static {
    System.loadLibrary(Core.NATIVE\_LIBRARY\_NAME);
}

public static void main(String\[\] args) {
    System.load("D:/Program Files/opencv/build/bin/opencv\_videoio\_ffmpeg411\_64.dll"); //加载opencv\_ffmpeg400\_64.dll，要不然是无法解码

    VideoCapture vc = new VideoCapture();
	boolean isOpen = vc.open("rtsp://192.168.104.130/user=admin&password=admin&channel=1");
	System.out.println("isOpen="+isOpen);
	Mat mat = new Mat();
	String winName = "showFrame";  //窗口名称
	int height = 600,width = 800;  //定义窗口宽和高
	HighGui.namedWindow(winName);
	HighGui.resizeWindow(winName, width, height);
	while(vc.read(mat)){	
		Mat dst = new Mat();
		Imgproc.resize(mat, dst, new Size(width,height));  // 重置窗口大小
		HighGui.imshow(winName, dst);  //显示窗口
		int key = HighGui.waitKey(1);  // 调用waitkey方法，否则无法显示
	}
	HighGui.destroyAllWindows();
	vc.release();
}

}

七、判断摄像头是否黑屏

 

import org.opencv.core.Mat; import java.io.File; import org.opencv.core.Core; import org.opencv.imgcodecs.Imgcodecs; import org.opencv.videoio.VideoCapture;

public class opencvTest {

static {
    System.loadLibrary(Core.NATIVE\_LIBRARY\_NAME);
}

private static void read\_camera() {
	VideoCapture camera = new VideoCapture("rtsp://192.168.104.130/user=admin&password=admin123&channel=1");
	Mat mat = new Mat();
	camera.read(mat);
	String filename = "test.png";
    Imgcodecs.imwrite(filename, mat);
	camera.release();
}

public static void main(String\[\] args) {
	System.load("D:/Users/Administrator/Desktop/opencv/build/bin/opencv\_videoio\_ffmpeg411\_64.dll");
	File file = new File("test.png");
	
	opencvTest.read\_camera();  //获取摄像头画面，并保存为test.png

    //如果文件大小不超过60000字节（60Kb），则判断摄像头黑屏
    if (file.length() > 60000) {
    	System.out.println("摄像头正常");
    }else {
    	System.out.println("摄像头黑屏");
    }
}
