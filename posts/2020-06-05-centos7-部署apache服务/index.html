Centos7 部署Apache服务

Jun 5, 2020
1 min read

1、安装Apache

#安装 [root@web ~]# yum -y install httpd

#防火墙开放http服务 [root@web ~]# firewall-cmd –permanent –add-service=http [root@web ~]# firewall-cmd –reload

#新建测试页面 [root@web ~]# echo “web page” > /var/www/html/index.html

#启动服务 [root@web ~]# systemctl start httpd [root@web ~]# systemctl enable httpd

#测试 使用浏览器访问web主机IP

2、修改网站目录

#新建网站目录及页面 [root@web ~]# mkdir /web [root@web ~]# echo “new web page” > /web/index.html

#修改主配置文件 [root@web ~]# vim /etc/httpd/conf/httpd.conf //修改以下两处位置 DocumentRoot “/web”
<Directory “/web”>

#重启httpd服务 [root@web ~]# systemctl restart httpd

#关闭selinux [root@web ~]# setenforce 0

#测试 使用浏览器访问web主机IP

3、新建虚拟主机

#新建虚拟主机配置文件，注意文件一定要是.conf的后缀 [root@web ~]# vim /etc/httpd/conf.d/test1.com.conf <VirtualHost *:80> ServerName www.test1.com DocumentRoot /web/test1/

[root@web ~]# vim /etc/httpd/conf.d/test2.com.conf Listen 81 //修改虚拟主机端口为81 <VirtualHost *:81> ServerName www.test2.com DocumentRoot /web/test2

#新建网站目录及文件 [root@web ~]# mkdir /web/test1 [root@web ~]# echo “test1 page” > /web/test1/index.html [root@web ~]# mkdir /web/test2 [root@web ~]# echo “test2 page” > /web/test2/index.html #防火墙开放端口 [root@web ~]# firewall-cmd –permanent –add-port=81/tcp [root@web ~]# firewall-cmd –reload

#重启httpd服务 [root@web ~]# systemctl restart httpd

#测试 修改客户端电脑host文件： 192.168.1.156 www.test1.com 192.168.1.156 www.test2.com 浏览器访问www.test1.com和www.test2.com:81（如果无法正常访问，检查网站目录权限，apache用户是否有读权限）
