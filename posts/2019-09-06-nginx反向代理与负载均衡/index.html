Nginx反向代理与负载均衡

Sep 6, 2019
2 min read

正向代理和反向代理的区别：

正向代理：代理服务器为客户端去收发请求，隐藏真实客户端

反向代理：代理服务器为服务端收发请求，隐藏真实服务端

需求一：隐藏真实网站服务器，由代理服务器去访问内部网站服务器，提高了内部服务器安全性

1、安装nginx

[root@server1 ~]# yum -y install epel-release //安装EPEL源 [root@server1 ~]# yum -y install nginx [root@server1 ~]# systemctl start nginx [root@server1 ~]# systemctl enable nginx

2、配置防火墙

[root@server1 ~]# firewall-cmd –permanent –add-service=http success [root@server1 ~]# firewall-cmd –reload success

3、配置反向代理

[root@server1 ~]# vim /etc/nginx/nginx.conf

注释掉默认主机

server {

listen 80 default_server;

listen [::]:80 default_server;

server_name _;

root /usr/share/nginx/html;

# Load configuration files for the default server block.

include /etc/nginx/default.d/*.conf;

location / {

}

error_page 404 /404.html;

location = /40x.html {

}

error_page 500 502 503 504 /50x.html;

location = /50x.html {

}

}

[root@server1 ~]# vim /etc/nginx/conf.d/proxy.conf

新建代理主机

server { listen 80; index index.html index.php;

location / { proxy_pass http://192.168.1.248 //指定被代理的主机IP include proxy_params; //包含代理参数文件 } }

[root@server1 ~]# vim /etc/nginx/proxy_params

配置代理参数

proxy_redirect default; proxy_set_header Host $http_host; proxy_set_header X-Real-IP $remote_addr; proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

proxy_connect_timeout 30; proxy_send_timeout 60; proxy_read_timeout 60;

proxy_buffer_size 32k; proxy_buffering on; proxy_buffers 4 128k; proxy_busy_buffers_size 256k; proxy_max_temp_file_size 256k;

[root@server1 ~]# systemctl reload nginx

 

需求二：实现负载均衡，由代理服务器去随机访问后端服务器

1、模拟后端多台服务器

#模拟三个站点，为了显示负载均衡效果，网站内容各不相同 [root@server1]# mkdir /usr/share/nginx/html/{node1,node2,node3} [root@server1]# echo “this is node1” > /usr/share/nginx/html/node1/index.html [root@server1]# echo “this is node2” > /usr/share/nginx/html/node2/index.html [root@server1]# echo “this is node3” > /usr/share/nginx/html/node3/index.html

#配置虚拟主机 [root@server1 ]# vim /etc/nginx/conf.d/site.conf server { listen 81; index index.html; root /usr/share/nginx/html/node1/; } server{ listen 82; index index.html; root /usr/share/nginx/html/node2/; } server{ listen 83; index index.html; root /usr/share/nginx/html/node3/; }

2、配置防火墙和selinux

#防火墙放行81，82，83端口 [root@server1 conf.d]# firewall-cmd –permanent –add-port=81/tcp success [root@server1 conf.d]# firewall-cmd –permanent –add-port=82/tcp success [root@server1 conf.d]# firewall-cmd –permanent –add-port=83/tcp success [root@server1 conf.d]# firewall-cmd –reload

#关闭selinux安全子系统，因为nginx使用80以外的端口会被selinux拒绝 [root@server1 ]# setenforce 0

3、配置反向代理实现负载均衡

[root@server1 ~]# vim /etc/nginx/nginx.conf

注释掉默认主机

server {

listen 80 default_server;

listen [::]:80 default_server;

server_name _;

root /usr/share/nginx/html;

# Load configuration files for the default server block.

include /etc/nginx/default.d/*.conf;

location / {

}

error_page 404 /404.html;

location = /40x.html {

}

error_page 500 502 503 504 /50x.html;

location = /50x.html {

}

}

[root@server1 ~]# vim /etc/nginx/conf.d/proxy.conf #新建upstream虚拟服务池，里面包含负载均衡使用到的所有主机 upstream node{ server 192.168.200.20:81; server 192.168.200.20:82; server 192.168.200.20:83; }

server { listen 80;

location / {
proxy\_pass http://node;   //代理指向upstream虚拟服务池
include proxy\_params;
}

}

[root@server1 ~]# vim /etc/nginx/proxy_params

配置代理参数

proxy_redirect default; proxy_set_header Host $http_host; proxy_set_header X-Real-IP $remote_addr; proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

proxy_connect_timeout 30; proxy_send_timeout 60; proxy_read_timeout 60;

proxy_buffer_size 32k; proxy_buffering on; proxy_buffers 4 128k; proxy_busy_buffers_size 256k; proxy_max_temp_file_size 256k;

[root@server1 ~]# systemctl reload nginx
