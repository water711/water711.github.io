搭建Zabbix高可用架构

Posted by on Mon, Sep 30, 2019

环境：

zabbix1：192.168.1.241（Master）

zabbix2：192.168.1.242（Slave）

zabbix：192.168.1.240（虚拟IP）

配置hosts文件

[root@Zabbix1 ~]# vim /etc/hosts 192.168.1.241 zabbix1 192.168.1.242 zabbix2 192.168.1.240 zabbix

一、安装Zabbix

#安装Zabbix仓库 [root@Zabbix1 ~]# rpm -Uvh https://repo.zabbix.com/zabbix/4.2/rhel/7/x86_64/zabbix-release-4.2-2.el7.noarch.rpm [root@Zabbix1 ~]# yum clean all

#安装Zabbix server，Web前端，agent，数据库 [root@Zabbix1 ~]# yum -y install zabbix-server-mysql zabbix-web-mysql zabbix-agent mariadb mariadb-server

#初始化并创建数据库 [root@Zabbix1 ~]# systemctl enable mariadb [root@Zabbix1 ~]# systemctl start mariadb [root@Zabbix1 ~]# mysql_secure_installation [root@Zabbix1 ~]# mysql -uroot -p MariaDB [(none)]> create database zabbix character set utf8 collate utf8_bin; MariaDB [(none)]> grant all privileges on zabbix.* to zabbix@localhost identified by ‘password’; MariaDB [(none)]> grant all privileges on zabbix.* to zabbix@192.168.1.240 identified by ‘password’; MariaDB [(none)]> quit;

#导入初始架构和数据到zabbix数据库（这步zabbix2不用操作） [root@Zabbix1 ~]# zcat /usr/share/doc/zabbix-server-mysql*/create.sql.gz | mysql -uzabbix -p zabbix

#配置Zabbix server [root@Zabbix1 ~]# sed -i ‘/# DBPassword=/cDBPassword=password’ /etc/zabbix/zabbix_server.conf #指定数据库密码

#配置前端PHP时区 [root@Zabbix1 ~]# sed -ri ‘/.*?php_value date/c\php_value date.timezone Asia/Shanghai’ /etc/httpd/conf.d/zabbix.conf //设置时区为上海

#关闭Selinux和配置防火墙 [root@Zabbix1 ~]# setenforce 0 //临时关闭Selinux [root@Zabbix1 ~]# sed -i /^SELINUX=/cSELINUX=disabled /etc/selinux/config //永久关闭Selinux [root@Zabbix1 ~]# firewall-cmd –permanent –add-service=http [root@Zabbix1 ~]# firewall-cmd –reload

#启动Zabbix相关服务 [root@Zabbix1 ~]# systemctl restart zabbix-server zabbix-agent httpd [root@Zabbix1 ~]# systemctl enable zabbix-server zabbix-agent httpd

 

二、部署keepalived

#两台机器安装keepalived [root@Zabbix1 ~]# yum -y install keepalived

[root@Zabbix1 ~]# vim /etc/keepalived/keepalived.conf ! Configuration File for keepalived

global_defs { router_id zabbix1 }

vrrp_script zabbix_chk { script “/etc/keepalived/zabbix_check.sh” interval 5 }

vrrp_instance VI_1 { state MASTER interface eth0 virtual_router_id 51 priority 100 advert_int 1 authentication { auth_type PASS auth_pass 1111 } virtual_ipaddress { 192.168.1.240 } track_script { zabbix_chk } notify_master “/etc/keepalived/zabbix.sh start” #指定当切换到master时，启动zabbix-server notify_backup “/etc/keepalived/zabbix.sh stop” #指定当切换到backup时，停止zabbix-server(防止两台机器同时运行zabbix-server,导致主从同步失败) notify_fault “/etc/keepalived/zabbix.sh stop” #当前节点出现故障,停止zabbix-server }

[root@Zabbix2 ~]# vim /etc/keepalived/keepalived.conf ! Configuration File for keepalived

global_defs { router_id zabbix2 }

vrrp_instance VI_1 { state BACKUP interface eth0 virtual_router_id 51 priority 50 advert_int 1 authentication { auth_type PASS auth_pass 1111 } virtual_ipaddress { 192.168.1.240 }

notify_master “/etc/keepalived/zabbix.sh start” #指定当切换到master时，启动zabbix-server notify_backup “/etc/keepalived/zabbix.sh stop” #指定当切换到backup时，停止zabbix-server notify_fault “/etc/keepalived/zabbix.sh stop” #当前节点出现故障,停止zabbix-server }

 

服务状态检测脚本（只配置在zabbix1上）

[root@Zabbix1 ~]# vim /etc/keepalived/zabbix_check.sh #!/bin/bash

zabbix_server_status1=$(ps -C zabbix_server –no-heading|wc -l)

if [ “${zabbix_server_status1}” = “0” ]; then systemctl restart zabbix_server.service sleep 2

zabbix_server_status2=$(ps -C zabbix_server –no-heading|wc -l) if [ “${zabbix_server_status2}” = “0” ]; then exit 1 fi fi

mariadb_status1=$(ps -C mysqld –no-heading|wc -l)

if [ “${mariadb_status1}” = “0” ]; then systemctl restart mariadb.service sleep 2

mariadb_status2=$(ps -C mysqld –no-heading|wc -l) if [ “${mariadb_status2}” = “0” ]; then exit 1 fi fi

httpd_status1=$(ps -C httpd –no-heading|wc -l)

if [ “${httpd_status1}” = “0” ]; then systemctl restart httpd.service sleep 2

httpd_status2=$(ps -C httpd –no-heading|wc -l) if [ “${httpd_status2}” = “0” ]; then exit 1 fi fi

[root@Zabbix1 ~]# chmod +x /etc/keepalived/zabbix_check.sh

 

notify脚本

[root@Zabbix1 ]# vim /etc/keepalived/zabbix.sh #!/bin/bash case “$1” in start) systemctl start zabbix-server ;; stop) systemctl stop zabbix-server ;; esac

[root@Zabbix1 ]# chmod +x /etc/keepalived/zabbix.sh

 

防火墙配置允许VRRP组播

[root@Zabbix1 ~]# firewall-cmd –direct –permanent –add-rule ipv4 filter INPUT 0 –in-interface eth0 –destination 224.0.0.18 –protocol vrrp -j ACCEPT [root@Zabbix1 ~]# firewall-cmd –reload

 

[root@Zabbix1 ~]# systemctl restart keepalived.service
[root@Zabbix1 ~]# systemctl enable keepalived.service

三、配置mariadb数据库主主同步

防火墙开放数据库端口

[root@Zabbix1 ~]# firewall-cmd –permanent –add-port=3306/tcp [root@Zabbix1 ~]# firewall-cmd –reload

 

zabbix1上操作

#备份原有配置 [root@Zabbix1 ~]# cp /etc/my.cnf{,.bak}

#编辑zabbix1的my.cnf文件 [root@Zabbix1 ~]# vim /etc/my.cnf [mysqld] datadir=/var/lib/mysql socket=/var/lib/mysql/mysql.sock symbolic-links=0

server-id=241 #保证每个服务器不一样，这里以IP最后一位来区分 log-bin=mysql-bin #开启bin日志 auto-increment-increment = 2 #步进值auto_imcrement。一般有n台主MySQL就填n auto-increment-offset = 1 #起始值。一般填第n台主MySQL。此时为第一台主MySQL binlog_format = mixed #设置binlog格式为mixed sync_binlog=1 #事务特性每1次事务提交 binlog-do-db=zabbix #同步的据库名 binlog-ignore-db=mysql #不同步数据库名 binlog-ignore-db=sys binlog-ignore-db=performance_schema binlog-ignore-db=information_schema replicate-do-db=zabbix #需要复制的数据库
replicate-ignore-db=mysql #不需要复制的数据库 replicate-ignore-db=sys replicate-ignore-db=performance_schema replicate-ignore-db=information_schema

[mysqld_safe] log-error=/var/log/mariadb/mariadb.log pid-file=/var/run/mariadb/mariadb.pid

#重启mariadb [root@Zabbix1 ~]# systemctl restart mariadb.service

#登录MariaDB控制台，授权同步用户，用于zabbix2上同步数据 [root@Zabbix1 ~]# mysql -uroot -p MariaDB [(none)]> GRANT REPLICATION SLAVE ON *.* TO ‘zabbix242’@‘192.168.1.242’ IDENTIFIED BY ‘zabbix242’; MariaDB [(none)]> FLUSH PRIVILEGES; MariaDB [(none)]> exit

#备份zabbix库 [root@Zabbix1 ~]# mysqldump -uroot -p zabbix –single-transaction –master-data=1 –flush-logs > zabbix.sql

#把备份文件拷贝到zabbix2机器上 [root@Zabbix1 ~]# scp zabbix.sql zabbix2:/root/

zabbix2上操作

#备份原有配置 [root@Zabbix2 ~]# cp /etc/my.cnf{,.bak}

#编辑zabbix2的my.cnf文件 [root@Zabbix2 ~]# vim /etc/my.cnf [mysqld] datadir=/var/lib/mysql socket=/var/lib/mysql/mysql.sock symbolic-links=0 server-id=242 #保证每个服务器不一样 log-bin=mysql-bin #开启bin日志 auto-increment-increment = 2 #步进值auto_imcrement。一般有n台主MySQL就填n auto-increment-offset = 2 #起始值。一般填第n台主MySQL。此时为第一台主MySQL binlog_format = mixed #设置binlog格式为mixed sync_binlog=1 #事务特性每1次事务提交 binlog-do-db=zabbix #同步的据库名 binlog-ignore-db=mysql #不同步数据库名 binlog-ignore-db=sys binlog-ignore-db=performance_schema binlog-ignore-db=information_schema replicate-do-db=zabbix #需要复制的数据库
replicate-ignore-db=mysql #不需要复制的数据库 replicate-ignore-db=sys replicate-ignore-db=performance_schema replicate-ignore-db=information_schema

[mysqld_safe] log-error=/var/log/mariadb/mariadb.log pid-file=/var/run/mariadb/mariadb.pid

#重启mariadb [root@Zabbix2 ~]# systemctl restart mariadb.service

 

#登录MariaDB控制台 [root@Zabbix2 ~]# mysql -uroot -p

#指定Master信息 MariaDB [(none)]> CHANGE MASTER TO MASTER_HOST=‘192.168.1.241’, MASTER_USER=‘zabbix242’, MASTER_PASSWORD=‘zabbix242’;

#导入zabbix1备份的数据库 MariaDB [(none)]> use zabbix MariaDB [zabbix]> set sql_log_bin=0; #临时关闭二进制日志 MariaDB [zabbix]> source /root/zabbix.sql

#启动SLAVE MariaDB [zabbix]> START SLAVE;

#检查主从复制是否配置成功（Slave_IO_Running和Slave_SQL_Running是否为YES） MariaDB [zabbix]> SHOW SLAVE STATUS\G;

五、zabbix前端文件及配置，主备同步（rsync +inotify）

#安装文件同步组件 [root@Zabbix1 ~]# yum -y install epel-release [root@Zabbix1 ~]# yum -y install rsync inotify-tools

 

#配置ssh秘钥登录 [root@Zabbix1 ~]# ssh-keygen //生成秘钥 [root@Zabbix1 ~]# ssh-copy-id zabbix2 //发送公钥到zabbix2，实现免密码远程登录

[root@Zabbix1 ~]# ssh-keygen //生成秘钥 [root@Zabbix1 ~]# ssh-copy-id zabbix1 //发送公钥到zabbix1，实现免密码远程登录

 

#网站及配置文件首次同步 [root@Zabbix1 ~]# rsync -rdvLzP –delete /usr/share/zabbix/ zabbix2:/usr/share/zabbix/ [root@Zabbix1 ~]# rsync -rdvLzP –delete /etc/zabbix/ zabbix2:/etc/zabbix/

 

#通过inotify监控文件系统变化并自动同步文件 [root@Zabbix1 ~]# vim rsync.sh source_path_1=/usr/share/zabbix/ source_path_2=/etc/zabbix/ log_file=/var/log/rsync_client.log

inotify_fun1(){ /usr/bin/inotifywait -mrq –timefmt ‘%Y/%m/%d-%H:%M:%S’ –format ‘%T %w %f’ \ -e modify,delete,create,move,attrib ${source_path_1} \ | while read file do rsync -rvLzP –delete ${source_path_1} zabbix2:${source_path_1}
done }

inotify_fun2(){ /usr/bin/inotifywait -mrq –timefmt ‘%Y/%m/%d-%H:%M:%S’ –format ‘%T %w %f’ \ -e modify,delete,create,move,attrib ${source_path_2} \ | while read file do rsync -rdvLzP –delete ${source_path_2} zabbix2:${source_path_2} done }

inotify_fun1 » ${log_file} 2>&1 & inotify_fun2 » ${log_file} 2>&1 &

 

#将脚本设置为开机启动 [root@Zabbix1 ~]# cp rsync.sh /etc/init.d/ [root@Zabbix1 ~]# chmod +x /etc/rc.d/rc.local [root@Zabbix1 ~]# echo “sh /etc/init.d/rsync.sh” » /etc/rc.d/rc.local
